#### sort()メゾットの使い方
クイックソートとヒープソートの組み合わせでソートされる。(イントロソート)
計算量はO(NlogN)

```rust
    // 整数
    let mut v = vec![3, 1, 4];
    v.sort();  // [1, 3, 4]
    v.reverse();//[4, 3, 1]
    
    //実数の時はsort_by
    let mut v = vec![3.5, 1.2, 4.1];
    v.sort_by(|a, b| a.partial_cmp(b).unwrap());  // [1.2, 3.5, 4.1]
    
    //タプル
    let mut v: Vec<(usize, usize)> = Vec::new();
    v.push((1, 2));
    v.push((2, 4));
    v.sort_by(|a, b| a.0.cmp(&b.0));//昇順でソート
    v.sort_by(|a, b| b.0.cmp(&a.0));//降順でソート
    
    
    // String
    let mut v = vec!["c", "a", "b"];
    v.sort();  // ["a", "b", "c"]
    
    // char
    let mut v = vec!['c', 'a', 'b'];
    v.sort();  // ['a', 'b', 'c']
```
