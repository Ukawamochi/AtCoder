insertion_sort(&mut board);

fn insertion_sort<T>(xs: &mut [T])
where
    T: PartialOrd + std::fmt::Debug,
{

    for i in 1..xs.len() {
        for j in (0..i).rev() {
            if xs[j] < xs[j + 1] {
                break;
            }
            xs.swap(j, j + 1);
        }
    }
}
