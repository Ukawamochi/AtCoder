# 二次元累積和

```rust
fn build_cum(grid: &Vec<Vec<usize>>) -> Vec<Vec<usize>> {
    let h = grid.len();
    let w = grid[0].len();
    let mut cum = vec![vec![0usize; w + 1]; h + 1];
    for i in 1..=h {
        for j in 1..=w {
            cum[i][j] = grid[i-1][j-1] + cum[i-1][j] + cum[i][j-1] - cum[i-1][j-1];
        }
    }
    cum
}

fn rect_sum(cum: &Vec<Vec<usize>>, h1: usize, w1: usize, h2: usize, w2: usize) -> usize {
    cum[h2][w2] - cum[h1-1][w2] - cum[h2][w1-1] + cum[h1-1][w1-1]
}


let cum = build_cum(&grid);
rect_sum(&cum, h1, w1, h2, w2);
```

## 解説

```rust
use proconio::input;
use proconio::marker::Chars;

fn build_cum(grid: &Vec<Vec<usize>>) -> Vec<Vec<usize>> {
    let h = grid.len();
    let w = grid[0].len();
    let mut cum = vec![vec![0usize; w + 1]; h + 1];
    for i in 1..=h {
        for j in 1..=w {
            cum[i][j] = grid[i-1][j-1] + cum[i-1][j] + cum[i][j-1] - cum[i-1][j-1];
        }
    }
    cum
}

fn rect_sum(cum: &Vec<Vec<usize>>, h1: usize, w1: usize, h2: usize, w2: usize) -> usize {
    cum[h2][w2] + cum[h1-1][w1-1] - cum[h1-1][w2] - cum[h2][w1-1]
}

fn main() {
    input! {
        h: usize,
        w: usize,
        k: usize,
        s: [Chars; h],
    }

    // Charを数値の二次元配列に変換する
    // sはVec<Vec<char>>
    let grid: Vec<Vec<usize>> = s.iter()
        .map(|row| row.iter().map(|&c| if c == '1' { 1 } else { 0 }).collect())
        .collect();

    // 累積和を構築
    // 入力=Vec<Vec<usize>>
    // 返り値=Vec<Vec<usize>>
    let cum = build_cum(&grid);

    // 全長方形を列挙して和がkのものを数える
    rect_sum(&cum, h1, w1, h2, w2);
}
```
