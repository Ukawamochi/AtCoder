# 文字列を配列に変換する

```rust
use proconio::input;
use proconio::marker::Chars;

fn main() {
    input! {
        n: usize,
        m: usize,
        // grid は Vec<Vec<char>> 
        grid: [Chars; n],
    }

    // Vec<Vec<char>> => Vec<Vec<usize>>
    let grid: Vec<Vec<usize>> = s.iter()
        .map(|row| row.iter().map(|&c| if c == '1' { 1 } else { 0 }).collect())
        .collect();
}
```
